Ejemplos Python Principiantes

Un programa que convierte grados Celsius a grados Fahrenheit.

            # Pedir al usuario ingresar la temperatura en grados Celsius
celsius = float(input("Ingresa la temperatura en grados Celsius: "))
# Convertir la temperatura de Celsius a Fahrenheit usando la fórmula: F = C * 9/5 + 32
fahrenheit = (celsius * 9/5) + 32
# Mostrar la temperatura en grados Fahrenheit
print(f"{celsius} grados Celsius equivalen a {fahrenheit} grados Fahrenheit.")

Un programa que genera números aleatorios y los ordena.

            import random
# Generar una lista de 10 números aleatorios entre 1 y 100
numeros = [random.randint(1, 100) for _ in range(10)]
# Ordenar los números de menor a mayor usando el método sort()
numeros.sort()
# Mostrar la lista ordenada
print("Números aleatorios ordenados:", numeros)

Un programa que calcule la media, mediana y moda de un conjunto de datos.

            import statistics
# Definir el conjunto de datos como una lista de números
datos = [5, 8, 10, 12, 5, 8, 10, 12, 15, 20]
# Calcular la media usando la función mean() del módulo statistics
media = statistics.mean(datos)
# Calcular la mediana usando la función median() del módulo statistics
mediana = statistics.median(datos)
# Calcular la moda usando la función mode() del módulo statistics
moda = statistics.mode(datos)
# Mostrar los resultados
print(f"Media: {media}")
print(f"Mediana: {mediana}")
print(f"Moda: {moda}")

Un programa que lea un archivo CSV y lo convierta en una lista de diccionarios.

datos.csv

            nombre,edad,ciudad
Juan,25,Medellín
María,30,Bogotá
Carlos,22,Cali 
          
            import csv
# Abrir el archivo CSV
with open("datos.csv") as archivo:
    # Leer los datos del archivo usando la función DictReader del módulo csv
    datos_csv = csv.DictReader(archivo)
    # Convertir los datos en una lista de diccionarios
    lista_datos = [fila for fila in datos_csv]
# Mostrar la lista de diccionarios resultante
print(lista_datos)

Un programa que realice web scraping de una página web y extraiga información específica.

            import requests
from bs4 import BeautifulSoup
# Hacer una petición GET a la página web deseada
url = "https://www.example.com"
respuesta = requests.get(url)
# Parsear el contenido HTML con BeautifulSoup
soup = BeautifulSoup(respuesta.content, "html.parser")
# Extraer la información deseada del contenido HTML usando métodos de BeautifulSoup
titulo = soup.title.text
parrafos = [p.text for p in soup.find_all("p")]
# Mostrar la información extraída
print("Título:", titulo)
print("Parrafos:", parrafos)

Un programa que se conecte a una base de datos y realice operaciones CRUD (crear, leer, actualizar y eliminar).

Este ejemplo asume que se utiliza una base de datos MySQL y que se tiene acceso a un servidor local con una base de datos llamada mi_bd y una tabla llamada mi_tabla.

            import mysql.connector
# Conectar a la base de datos
conexion = mysql.connector.connect(
    host="localhost",
    user="mi_usuario",
    password="mi_contraseña",
    database="mi_bd"
)
# Crear un cursor para ejecutar consultas SQL
cursor = conexion.cursor()
# Ejecutar una consulta SQL para crear una nueva entrada en la tabla
consulta = "INSERT INTO mi_tabla (nombre, edad) VALUES (%s, %s)"
valores = ("Juan", 25)
cursor.execute(consulta, valores)
# Ejecutar una consulta SQL para leer las entradas de la tabla
consulta = "SELECT * FROM mi_tabla"
cursor.execute(consulta)
# Leer los resultados de la consulta y mostrarlos
resultados = cursor.fetchall()
for fila in resultados:
    print(f"Nombre: {fila[1]}, Edad: {fila[2]}")
# Ejecutar una consulta SQL para actualizar una entrada de la tabla
consulta = "UPDATE mi_tabla SET edad = %s WHERE nombre = %s"
valores = (30, "Juan")
cursor.execute(consulta, valores)
# Ejecutar una consulta SQL para eliminar una entrada de la tabla
consulta = "DELETE FROM mi_tabla WHERE nombre = %s"
valores = ("Juan",)
cursor.execute(consulta, valores)
# Confirmar los cambios y cerrar la conexión
conexion.commit()
conexion.close()

Aplicación web para permitir a los usuarios registrarse y autenticarse.

            from flask import Flask, render_template, request, redirect, url_for, session
import bcrypt
app = Flask(__name__)
app.secret_key = "mi_clave_secreta"
# Página de inicio
@app.route("/")
def inicio():
    if "usuario" in session:
        return f"Bienvenido, {session['usuario']}!"
    else:
        return render_template("inicio.html")
# Página de registro
@app.route("/registro", methods=["GET", "POST"])
def registro():
    if request.method == "POST":
        # Obtener el nombre de usuario y la contraseña ingresados por el usuario
        nombre_usuario = request.form["usuario"]
        contrasena = request.form["contrasena"].encode("utf-8")
        # Encriptar la contraseña utilizando bcrypt
        salt = bcrypt.gensalt()
        contrasena_encriptada = bcrypt.hashpw(contrasena, salt)
        # Agregar el nuevo usuario y la contraseña encriptada a la base de datos
        # (Este código se omite para simplificar el ejemplo)
        # Iniciar sesión para el nuevo usuario
        session["usuario"] = nombre_usuario
        return redirect(url_for("inicio"))
    else:
        return render_template("registro.html")
# Página de inicio de sesión
@app.route("/iniciar_sesion", methods=["GET", "POST"])
def iniciar_sesion():
    if request.method == "POST":
        # Obtener el nombre de usuario y la contraseña ingresados por el usuario
        nombre_usuario = request.form["usuario"]
        contrasena = request.form["contrasena"].encode("utf-8")
        # Obtener la contraseña encriptada del usuario desde la base de datos
        # (Este código se omite para simplificar el ejemplo)
        # Verificar si la contraseña ingresada es correcta
        if bcrypt.checkpw(contrasena, contrasena_encriptada):
            session["usuario"] = nombre_usuario
            return redirect(url_for("inicio"))
        else:
            return render_template("iniciar_sesion.html", mensaje="Usuario o contraseña incorrectos")
    else:
        return render_template("iniciar_sesion.html")
# Página de cierre de sesión
@app.route("/cerrar_sesion")
def cerrar_sesion():
    session.pop("usuario", None)
    return redirect(url_for("inicio"))
